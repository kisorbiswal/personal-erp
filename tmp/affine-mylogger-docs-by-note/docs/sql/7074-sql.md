CREATE OR REPLACE FUNCTION amcInstalationUpdate( ) RETURNS boolean AS
$BODY$
DECLARE
    r      _amc_information%ROWTYPE;
	r1   _amc_instalment%ROWTYPE;
    _no integer := 1;
	cid  _company%ROWTYPE;
BEGIN
For cid in select id from _company
loop 
	_no:=1;
	FOR r IN
		SELECT id
		FROM   _amc_information  where _company_id = cid.id  order by created_Date
	LOOP
		 For r1 In Select id from _amc_instalment where _amc = r.id  order by created_date 
		 Loop 
		   UPDATE _amc_instalment set _installment_no = _no where id = r1.id;
		   _no := _no + 1;
		 END Loop; 
	END LOOP;
End loop;
return true;
END;
 $BODY$
  LANGUAGE plpgsql VOLATILE
  COST 100;
ALTER FUNCTION amcInstalationUpdate()
  OWNER TO meta;
END;
select amcInstalationUpdate() from _company;
