Validation of table and dropdown function validateTables( ) {
	var isSuccess=true;
	$j(".mustHaveRow").each(function(){
		var $this = $j(this);
		
		var $divError = $this.find("div.errorTable");
		if($divError.length==1){
		$divError.remove();
		}
		var rows= $this[0].rows;
		if(rows.length<2){
			isSuccess=false;
			$j("<div/>").addClass("errorTable").text("Must have atleast one row.").appendTo($this.find("tr:first").find("th:first")).click(function(){$j(this).hide();});
		}
	});

	return isSuccess;
}

function validateDropdowns( ) {
	var isSuccess=true;
	$j(".mustSelect").each(function(){
		var $this = $j(this);
		
		var $divError = $this.parent().find("div.errorTable");
		if($divError.length==1){
		$divError.remove();
		}
		var selectedVal= $this.val();
		if(selectedVal=="empty"){
			isSuccess=false;
			$j("<div/>").addClass("errorTable").text("Must select one option.").appendTo($this.parent()).click(function(){$j(this).hide();});
		}
	});

	return isSuccess;
}
