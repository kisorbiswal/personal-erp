private Map<String, String> getInfoByIP(String ip){ 
          try { 
          String apiLocation="http://api.ipinfodb.com/v3/ip-city/?key=ec542ed17fbb0132019c7a76f21be325cc4c3bd61e893b7f5e693d02a7778646&ip="+ip+"&format=json"; 
               URL url=new URL(apiLocation); 
               InputStream inputStream = url.openStream(); 
               ByteArrayOutputStream buffer = new ByteArrayOutputStream(); 
 
               int nRead; 
               byte[] data = new byte[16384]; 
 
               while ((nRead = inputStream.read(data, 0, data.length)) != -1) { 
                 buffer.write(data, 0, nRead); 
               } 
 
               buffer.flush(); 
               String rawInfo=new String(buffer.toByteArray()); 
               return parseJSON(rawInfo); 
          } catch (Exception e) { 
               e.printStackTrace(); 
          } 
          return null; 
     } 
      
     private Map<String, String> parseJSON(String jsonString){ 
          if(jsonString==null||jsonString.length()<1){ 
               return null; 
          } 
           
          jsonString=jsonString.substring(1,jsonString.length()-1); 
          String[] pairs = jsonString.split(","); 
          Map<String, String> infoMap=new HashMap<String, String>(); 
          for(String s:pairs){ 
               String[] pair = s.split(":"); 
               infoMap.put(pair[0].trim().replaceAll("\"", ""), pair[1].trim().replaceAll("\"", "")); 
          } 
          return infoMap; 
     }
