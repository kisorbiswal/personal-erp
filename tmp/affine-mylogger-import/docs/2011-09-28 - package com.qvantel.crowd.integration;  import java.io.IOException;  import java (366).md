package com.qvantel.crowd.integration;  import java.io.IOException;  import javax.servlet.RequestDispatcher; import javax.servlet.ServletException; import javax.servlet.http.HttpServletRequest; import javax.servlet.http.HttpServletResponse;  import org.springframework.security.authentication.AbstractAuthenticationToken; import org.springframework.security.authentication.UsernamePasswordAuthenticationToken; import org.springframework.security.core.Authentication; import org.springframework.security.core.AuthenticationException; import org.springframework.security.core.context.SecurityContextHolder; import org.springframework.security.web.authentication.UsernamePasswordAuthenticationFilter; import org.springframework.security.web.savedrequest.DefaultSavedRequest;  import com.atlassian.crowd.integration.authentication.ValidationFactor; import com.atlassian.crowd.integration.exception.InvalidTokenException; import com.atlassian.crowd.integration.http.HttpAuthenticator; import com.atlassian.crowd.integration.springsecurity.CrowdSSOAuthenticationDetails; import com.atlassian.crowd.integration.springsecurity.CrowdSSOAuthenticationToken; import com.atlassian.crowd.integration.springsecurity.RequestToApplicationMapper; import com.atlassian.crowd.integration.springsecurity.user.CrowdUserDetails;  public class SSOAuthenticationProcessingFilter extends 		UsernamePasswordAuthenticationFilter {  	private HttpAuthenticator httpAuthenticator; 	private RequestToApplicationMapper requestToApplicationMapper; 	private String success;  	protected boolean requiresAuthentication(HttpServletRequest request, 			HttpServletResponse response) { 		boolean usernamePasswordAuthentication = super.requiresAuthentication( 				request, response);  		if (!usernamePasswordAuthentication) { 			AuthenticationException exp = null; 			Authentication authenticatedToken = null; 			try { 				String uri = request.getRequestURL().toString(); 				String tokenString = this.httpAuthenticator.getToken(request); 				if (uri.contains("logout")) { 					tokenString = null; 				} 				CrowdSSOAuthenticationToken crowdAuthRequest = new CrowdSSOAuthenticationToken( 						tokenString); 				doSetDetails(request, crowdAuthRequest);  				authenticatedToken = getAuthenticationManager().authenticate( 						crowdAuthRequest);  			} catch (InvalidTokenException e) { 			} catch (AuthenticationException e) { 				exp = e; 			}  			if (authenticatedToken == null) { 				SecurityContextHolder.clearContext();  				try { 					// If there is any exception forward to application context 					// url, because response.redirect is not working 					String uri = request.getRequestURL().toString(); 					if (!isAppUrl(request) 							&& (uri.contains("view") || uri.contains("htm"))) { 						RequestDispatcher rd = request 								.getRequestDispatcher("/"); 						rd.forward(request, response); 					} 				} catch (Exception e) { 					e.printStackTrace(); 				} 			} else { 				SecurityContextHolder.getContext().setAuthentication( 						authenticatedToken); 				try { 					onSuccessfulAuthentication(request, response, 							authenticatedToken);  				} catch (IOException e) { 				} 			} 		} 		return usernamePasswordAuthentication; 	}  	private boolean isAppUrl(HttpServletRequest request) { 		String uri = request.getRequestURL().toString(); 		String context = request.getContextPath(); 		boolean isAppUrl = false; 		if (uri.endsWith(context) || uri.endsWith(context + "/")) { 			isAppUrl = true; 		} 		return isAppUrl; 	}  	@Override 	protected void successfulAuthentication(HttpServletRequest request, 			HttpServletResponse response, Authentication authResult) 			throws IOException, ServletException { 		onSuccessfulAuthentication(request, response, authResult); 		super.successfulAuthentication(request, response, authResult); 		System.out.println("Real type of  authResult.getPrincipal(): " 				+ authResult.getPrincipal().getClass().getName()); 		if (authResult.getPrincipal() instanceof CrowdUserDetails) { 			System.out.println("Principal after authentication" 					+ ((CrowdUserDetails) authResult.getPrincipal()) 							.getFullName()); 		} 	}  	@Override 	protected void unsuccessfulAuthentication(HttpServletRequest request, 			HttpServletResponse response, AuthenticationException failed) 			throws IOException, ServletException { 		onUnsuccessfulAuthentication(request, response, failed); 		super.unsuccessfulAuthentication(request, response, failed); 	}  	protected void setDetails(HttpServletRequest request, 			UsernamePasswordAuthenticationToken authRequest) { 		doSetDetails(request, authRequest); 	}  	protected void doSetDetails(HttpServletRequest request, 			AbstractAuthenticationToken authRequest) { 		String application; 		if (this.requestToApplicationMapper != null) { 			DefaultSavedRequest savedRequest = (DefaultSavedRequest) request 					.getSession().getAttribute( 							"SPRING_SECURITY_SAVED_REQUEST_KEY"); 			String path; 			if (savedRequest != null) 				path = savedRequest.getRequestURI().substring( 						savedRequest.getContextPath().length()); 			else { 				path = request.getRequestURI().substring( 						request.getContextPath().length()); 			}  			application = this.requestToApplicationMapper.getApplication(path); 		} else { 			application = this.httpAuthenticator.getClientProperties() 					.getApplicationName(); 		}  		ValidationFactor[] validationFactors = this.httpAuthenticator 				.getValidationFactors(request);  		authRequest.setDetails(new CrowdSSOAuthenticationDetails(application, 				validationFactors)); 	}  	protected void onSuccessfulAuthentication(HttpServletRequest request, 			HttpServletResponse response, Authentication authResult) 			throws IOException { 		System.out 				.println("On SuccessfulAuthentication Method [SSOAuthenticationProcessingFilter"); 		System.out 				.println("AuthResult InstanceOf CrowdSSOAuthenticationToken ...." 						+ Boolean 								.toString(authResult instanceof CrowdSSOAuthenticationToken)); 		System.out.println("AuthResult Credentials ....." 				+ authResult.getCredentials()); 		if (((authResult instanceof CrowdSSOAuthenticationToken)) 				&& (authResult.getCredentials() != null)) 			try { 				this.logger.info("[  Setting Crowd SSO Token  ] ...... ");  				this.httpAuthenticator.setPrincipalToken(request, response, 						authResult.getCredentials().toString()); 				this.logger.info("The real type of getPrincipal()" 						+ ((CrowdSSOAuthenticationToken) authResult) 								.getPrincipal().getClass().getName()); 				// Redirect to success 				String uri = request.getRequestURL().toString(); 				// if (uri.equals(applicationUrl)) { 				if (isAppUrl(request)) { 					// RequestDispatcher rd = request 					// .getRequestDispatcher("/home.view"); 					// rd.forward(request, response); 					response.sendRedirect(success); 				} 			} catch (Exception e) { 				this.logger.error("Unable to set Crowd SSO token", e); 			} 	}  	protected void onUnsuccessfulAuthentication(HttpServletRequest request, 			HttpServletResponse response, AuthenticationException failed) 			throws IOException { 		try { 			this.httpAuthenticator.logoff(request, response); 		} catch (Exception e) { 			this.logger.error("Unable to unset Crowd SSO token", e); 		} 	}  	public void setHttpAuthenticator(HttpAuthenticator httpAuthenticator) { 		this.httpAuthenticator = httpAuthenticator; 	}  	public void setRequestToApplicationMapper( 			RequestToApplicationMapper requestToApplicationMapper) { 		this.requestToApplicationMapper = requestToApplicationMapper; 	}  	public String getSuccess() { 		return success; 	}  	public void setSuccess(String success) { 		this.success = success; 	}  }
